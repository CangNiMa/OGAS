#Statue
building_gaudi_statue = {
    building_group = bg_architect
    icon = "gfx/interface/icons/building_icons/building_gaudi_statue.dds"
    expandable = no
    downsizeable = no

    potential = {
        total_urbanization >= 500
        owner = {
            has_variable = building_gaudi_statue_blueprint_var
        }
        NOT = {
            owner = {
                any_scope_state = {
                    NOT = { this = root }
                    any_scope_building = {
                        is_building_type = building_gaudi_statue
                    }
                }
            }
        }
    }

    possible = {
        total_urbanization >= 500
        owner = {
            has_variable = building_gaudi_statue_blueprint_var
        }
    }

    can_build_private = {
        always = no
    }

    production_method_groups = {
        pmg_gaudi_statue
    }

    required_construction = construction_cost_medium

    background = "gfx/interface/icons/building_icons/backgrounds/building_panel_bg_monuments.dds"
}
