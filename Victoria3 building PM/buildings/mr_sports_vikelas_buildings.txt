	}

		pmg_vikelas_sports_cricket

		pmg_vikelas_sports_football
		pmg_vikelas_sports_basketball
		pmg_vikelas_sports_grounds
	}

	icon = "gfx/interface/icons/building_icons/building_vikelas_sports.dds"

	required_construction = construction_cost_low

	}

	}

	can_build_private = {
		always = no
	}

	ai_value = 100

	background = "gfx/interface/icons/building_icons/backgrounds/building_panel_bg_monuments.dds"
}
