active_workforce_employment_pm_manager = {
	name = "active_workforce_employment_pm_manager"
	desc = "active_workforce_employment_pm_manager_desc"

	effect = {
		if = {
			limit = {
				root.var:cnm_workforce_employment_pm_manager = 0
			}
			set_variable = {
				name = cnm_workforce_employment_pm_manager
				value = 1
			}
		}
		else_if = {
			limit = {
				root.var:cnm_workforce_employment_pm_manager = 1
			}
			set_variable = {
				name = cnm_workforce_employment_pm_manager
				value = 0
			}
		}
	}
}

	name = "balance_tolerance_pm_manager_switch"
	desc = "balance_tolerance_pm_manager_switch_desc"

	effect = {
		if = {
			limit = {
				root.var:cnm_upgrade_tolerance_pm_manager = pm_upgrade_tolerance_high
			}
			set_variable = {
				name = cnm_upgrade_tolerance_pm_manager
				value = pm_upgrade_tolerance_low
			}
		}
		else_if = {
			limit = {
				root.var:cnm_upgrade_tolerance_pm_manager = pm_upgrade_tolerance_low
			}
			set_variable = {
				name = cnm_upgrade_tolerance_pm_manager
				value = pm_upgrade_tolerance_medium

			}

				name = cnm_upgrade_tolerance_pm_manager
				value = pm_upgrade_tolerance_high
			}
		}
	}
}

		}
	}
}
pm_manage_amount_sub = {
	name = "pm_manage_amount_sub"
	desc = "pm_manage_amount_sub_desc"

	effect = {
		if = {
			limit = {
				root.var:cnm_pm_manage_amount > 1
			}
			change_variable = {
				name = cnm_pm_manage_amount
				subtract = 1
			}
		}
	}
}
